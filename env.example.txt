# Supabase Configuration
# Copy isi file ini ke file .env di root project dan isi dengan nilai yang sesuai

# URL Supabase Project (dapat ditemukan di Supabase Dashboard > Settings > API)
VITE_SUPABASE_URL=your_supabase_project_url

# Supabase Anon/Public Key (dapat ditemukan di Supabase Dashboard > Settings > API)
VITE_SUPABASE_ANON_KEY=your_supabase_anon_key

# Supabase Service Role Key (Opsional - hanya untuk admin operations seperti create user)
# WARNING: Jangan expose key ini di client-side code!
# Dapat ditemukan di Supabase Dashboard > Settings > API
VITE_SUPABASE_SERVICE_ROLE_KEY=your_service_role_key
